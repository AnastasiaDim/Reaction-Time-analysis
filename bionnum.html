split_data <- list()
start_index <- 1

  end_index <- break_points[i]

plot_reaction_times <- function(data, segment_index) {
  mean_rts <- data %>%
    group_by(RATIO, MOVEMENT, NUMBER_1, NUMBER_2) %>%
    summarise(mean_rts = mean(`response in ms`), .groups = 'drop')
  
  ggplot(mean_rts, aes(x = factor(MOVEMENT), y = mean_rts, fill = factor(RATIO))) +
    geom_boxplot() +
    geom_point(position = position_jitter(width = 0.2), size = 2, alpha = 0.5) +
    scale_fill_brewer(palette = "PuBu") +  
    theme_bw() +
    labs(title = paste("Segment", segment_index, "Reaction Times"),
         x = "Movement", y = "Reaction Times")
}


plot_percentage_correct <- function(data, segment_index) {
  data_percentage <- data %>%
    group_by(MOVEMENT, RATIO) %>%
    summarise(percentage = mean(response.corr) * 100, .groups = 'drop')
  
  ggplot(data_percentage, aes(x = factor(MOVEMENT), y = percentage, fill = factor(RATIO))) +
    geom_bar(stat = "identity", position = "dodge") +
    scale_fill_brewer(palette = "BuPu") +
    theme_bw() +
    labs(title = paste("Segment", segment_index, "Percentage of Correct Responses"),
         x = "Movement", y = "Percentage of Correct Responses", fill = "Ratio")
}
